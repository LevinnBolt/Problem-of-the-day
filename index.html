11-03-2025 PROBLEM OF THE DAY

PROBLEM-1


SELECT UL.LISTENING_ID, UL.USER_ID, SPH.PLAY_COUNT 
FROM USER_LISTENING UL  
JOIN SONG_PLAY_HISTORY SPH 
ON UL.SONG_ID = SPH.SONG_ID;

SELECT UL.SONG_ID, SUM(UL.DURATION) AS TOTAL_DURATION 
FROM USER_LISTENING UL 
JOIN SONG_PLAY_HISTORY SPH 
ON UL.SONG_ID = SPH.SONG_ID 
GROUP BY UL.SONG_ID; 

SELECT UL.SONG_ID, SUM(SPH.PLAY_COUNT) AS TOTAL_PLAY_COUNT 
FROM USER_LISTENING UL 
JOIN SONG_PLAY_HISTORY SPH 
ON UL.SONG_ID = SPH.SONG_ID 
GROUP BY UL.SONG_ID;
